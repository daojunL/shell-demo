# integer comparison 

-eq - is equal to - if [ "$a" - eq "$b" ]
-ne - is not equal to - if [ "$a" -ne "$b" ]
-gt - is greater than 
-lt - is less than 
-le - is less than or equal to 
< - is less than 
<= - is less than or equal to 
> - is greater than 
>= - is greater than or equal to 

# string comparison 
= - is equal to - if [ "$a" = "$b"]

!= is not equal to 
< is less than 
> is greater than 
-z - string is null, that is, has zero length
